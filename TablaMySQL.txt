CREATE TABLE `usuarios` (
	Id BIGINT(20) PRIMARY KEY,
	Keys420 VARCHAR(34) NOT NULL,
	Creditos INT(36) NOT NULL,
	Rango VARCHAR(20) NOT NULL DEFAULT 'Cliente',
	UserTelegram VARCHAR(35) NOT NULL DEFAULT 'No asignado',
	Usando VARCHAR(2) NOT NULL DEFAULT 'no',
	FechaCrea DATETIME NOT NULL, 
	AntiSpam DECIMAL(10, 2) DEFAULT 60.0
);
